#include <Servo.h>

Servo myServo;  // create servo object

const int potPin = A0;   // Potentiometer pin
int potValue;            // To store raw pot value
int servoAngle;          // To store mapped angle

void setup() {
  myServo.attach(9);   // Servo signal pin to D9
  Serial.begin(9600);
  Serial.println("Servo Position Control with Potentiometer");
}

void loop() {
  potValue = analogRead(potPin);        // Read potentiometer (0-1023)
  servoAngle = map(potValue, 0, 1023, 0, 180);  // Map to 0-180 degrees
  myServo.write(servoAngle);            // Move servo
  delay(15);                            // Small delay for smooth movement

  // Print values on Serial Monitor
  Serial.print("Pot Value: ");
  Serial.print(potValue);
  Serial.print(" | Servo Angle: ");
  Serial.println(servoAngle);
}
