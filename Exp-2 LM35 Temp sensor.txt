#include <Wire.h>
#include <LiquidCrystal_I2C.h>

// Create LCD object with I2C address 0x27 and 16x2 size
LiquidCrystal_I2C lcd(0x27, 16, 2);

// LM35 sensor connected to analog pin A0
const int lm35Pin = A0;

void setup() {
  // Initialize Serial Monitor
  Serial.begin(9600);

  // Initialize LCD
  lcd.begin(16, 2);
  lcd.backlight();

  Serial.println("LM35 Temperature Sensor");
}

void loop() {
  // Read analog value from LM35
  int sensorValue = analogRead(lm35Pin);

  // Convert to voltage (5V Arduino has 10-bit ADC => 0-1023 steps)
  float voltage = (sensorValue * 5.0) / 1023.0;

  // LM35 gives 10mV per °C => Temp = (voltage * 100)
  float temperatureC = voltage * 100.0;

  // Print on Serial Monitor
  Serial.print("Temperature: ");
  Serial.print(temperatureC);
  Serial.println(" °C");

  // Display on LCD
  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print("Temperature:");
  lcd.setCursor(0, 1);
  lcd.print(temperatureC);
  lcd.print((char)223); // Degree symbol
  lcd.print("C");

  delay(1000); // Update every second
}
